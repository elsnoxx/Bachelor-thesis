@startuml
skinparam style strictuml
skinparam sequenceMessageAlign center

actor "Hráč (Hráč 1 + Hráč 2)" as Player
participant "React Frontend\n(EnergyBattleGame)" as FE
participant "BLE Service\n(Senzor Sensetio)" as BLE
participant "SignalR Hub\n(GameHub)" as Hub
participant "Backend Service\n(EnergyBattleService)" as Backend
database "PostgreSQL\n(via DbWriteQueue)" as DB

== 1. Inicializace a povinné připojení ==

Player -> FE : Vstup do místnosti
FE -> FE : Kontrola JWT Tokenu
FE -> Hub : Connect (Token)
FE -> Hub : JoinRoom(roomId)

Player -> FE : Připojení senzoru Sensetio
FE -> BLE : Vyhledat (Web Bluetooth API)
BLE -->> FE : Připojeno (isConnected = true)

== 2. Individuální kalibrace (10 s) ==

Player -> FE : Start hry
FE -> Hub : StartCalibration(roomId)
loop Každých 100ms
    BLE -> FE : Surová hodnota EDA (v_raw)
    FE -> FE : Normalizace: v_norm = (v_raw / 65535) * 1000
    FE -> Hub : SendData(v_norm)
    Hub -> Backend : ProcessCalibration(email, v_norm)
end
Backend -> Backend : Vypočti individuální baseline (klidová hladina)

== 3. Bojová fáze (Herní smyčka) ==

loop Každých 100ms
    BLE -> FE : Surová hodnota EDA (v_raw)
    FE -> FE : v_norm = (v_raw / 65535) * 1000
    FE -> Hub : SendEnergyData(roomId, v_norm)
    Hub -> Backend : ProcessInput(roomId, email, v_norm)

    group Výpočet energie na serveru
        alt Relaxační nabíjení: v_norm <= baseline
            Backend -> Backend : Plynulý nárůst energie (Tonická složka)
        end
        
        alt Aktivační bonus: Detekce prudkého nárůstu (SCR pík)
            Backend -> Backend : Okamžitý bonus +15% energie
        end
    end

    Backend -> DB : QueueBioFeedbackAsync(v_norm)
    Backend --> Hub : ReceiveGameState (Energy, HP soupeře)
    Hub -> FE : ReceiveGameState(data)
    FE -> Player : Aktualizuj UI (Energy bar, Health, Ready to Fire)
end

== 4. Akce: Výstřel (Fire) ==

Player -> FE : Klikne na "Vystřelit"
note right of FE : Povoleno v UI pouze při Energy >= 100%

FE -> Hub : FireCannon(roomId)
Hub -> Backend : Fire(roomId, shooterEmail)

alt Validace na serveru: Energy >= 100%
    Backend -> Backend : Reset útočníka: Energy = 0
    Backend -> Backend : Poškození soupeře: HP -= 20
    
    alt Zdraví soupeře <= 0
        Backend -> Backend : IsFinished = true
        Backend -> DB : QueueGameResultAsync(stats)
        Backend --> Hub : ReceiveGameState (isGameOver: true, Winner)
        FE -> Player : Zobraz konečný výsledek (Vítězství/Prohra)
    else Boj pokračuje
        Backend --> Hub : ReceiveGameState (Aktualizované HP)
        Hub -> FE : ReceiveGameState(data)
    end
end

@enduml
